# Load raw data and do little preprocessing.

def transformCommas(line):
    out = ''
    insideQuote = False
    for c in line:
        if c == '"':
            insideQuote = not insideQuote
            out += ''
        elif insideQuote == True and c == ',':
            out += ''
        else:
            out += c
    return out

def lineReplacements(line):
    line = line.replace('%', '')
    line = line.replace('$', '')
    return line

f = open("data/raw_data_all.csv", "rb")
replaced = (lineReplacements(transformCommas(line)) for line in f)
rawData = numpy.loadtxt(replaced ,delimiter=",",skiprows=0, dtype=str)

def weight_array(ar, weights): # http://stackoverflow.com/questions/21844024/weighted-percentile-using-numpy
     zipped = zip(ar, weights)
     weighted = []
     for i in zipped:
         for j in range(i[1]):
             weighted.append(i[0])
     return weighted
